        function initialiserReseau(idJoueur) {
            const reseauId = "rl2d-" + idJoueur.toLowerCase().replace(/[^a-z0-9]/g, "");
            try {
                peer = new Peer(reseauId);

        function envoyerInvitation() {
            let amiId = document.getElementById('invite-input').value.trim();
            if(!amiId) return alert("Tape le pseudo de ton ami !");
            const cibleId = "rl2d-" + amiId.toLowerCase().replace(/[^a-z0-9]/g, "");

            conn = peer.connect(cibleId); jeSuisHote = true; configurerEvenementsConnexion();

        function configurerEvenementsConnexion() {
            if (!conn) return;
            conn.on('open', () => { if(jeSuisHote) conn.send({ type: 'INVITATION', de: pseudo }); });
            conn.on('data', (data) => {
                if (data.type === 'INVITATION') {
                    document.getElementById('invite-msg').innerText = `${data.de} vous invite à disputer un match en ligne !`;
                    document.getElementById('invite-pop-up').style.display = "block";
                }
                else if (data.type === 'REPONSE_INVIT') {
                    if (data.accepte) lancerJeu(true);
                    else { alert("L'invitation a été refusée."); document.getElementById('my-online-id').innerText = pseudo; }
                }
                else if (data.type === 'SYNC_INPUTS') {
                    let cible = jeSuisHote ? bot : joueur;
                    cible.x = data.x; cible.y = data.y; cible.angle = data.angle; cible.turbo = data.turbo;
                    cible.skin = data.skin; cible.boostColor = data.boostColor;
                }
                else if (data.type === 'SYNC_BALL') {
                    if (!jeSuisHote) {
                        ball.x = data.x; ball.y = data.y; scoreBleu = data.sb; scoreRouge = data.sr; tempsRestant = data.temps;
                        document.getElementById('sb').innerText = pseudo + " : " + scoreBleu;
                        document.getElementById('sr').innerText = "ADVERSE : " + scoreRouge;
                    }
                }
                else if (data.type === 'BUT') { declencherButLocal(data.texte, data.color); }
            });
            conn.on('close', () => { alert("Déconnecté."); quitterMatch(); });
        }

        function gererAuthentification() {
            const user = document.getElementById('auth-username').value.trim();
            const pass = document.getElementById('auth-password').value;
            const errorBox = document.getElementById('auth-error');
            if (!user || !pass) { errorBox.innerText = "Remplis tous les champs !"; errorBox.style.display = "block"; return; }

            let tousLesComptes = JSON.parse(localStorage.getItem('rl2d_comptes')) || {};
            let compteKey = user.toLowerCase();

            if (modeInscription) {

        }

        // --- MANETTE GAMEPAD : POLLING NATION ---
        function scanGamepad() {
            if (peripheriqueChoisi !== 'manette') return;
            const gamepads = navigator.getGamepads ? navigator.getGamepads() : [];
            const gp = gamepads[0];
            if (!gp) return;

            // Sticks de visée directionnelle
            let lx = gp.axes[0], ly = gp.axes[1];
            if (Math.abs(lx) > 0.1 || Math.abs(ly) > 0.1) {
                let cible = jeSuisHote || !estEnLigne ? joueur : bot;
                cible.angle = Math.atan2(ly, lx);
            }
            // Boutons d'actions
            keys.avant = gp.buttons[0].pressed; // Bouton A / Croix
            keys.turbo = gp.buttons[1].pressed; // Bouton B / Rond
        }

        // --- BOUCLE PRINCIPALE ---
        function update() {
            if (jeuEnCours) {
                boostPads.forEach(p => { if (!p.active) { p.cooldown--; if (p.cooldown <= 0) p.active = true; } });
                
                scanGamepad(); // Récupère l'état de la manette si active

                let maVoiture = jeSuisHote || !estEnLigne ? joueur : bot;

                // Calcul orientation selon périphérique standard (Clavier / Souris)
                if(peripheriqueChoisi === 'clavier' || peripheriqueChoisi === 'souris') {
                    let dxMouse = mouseX - (canvas.width / 2), dyMouse = mouseY - (canvas.height / 2);
                    maVoiture.angle = Math.atan2(dyMouse, dxMouse);
                } 
                else if(peripheriqueChoisi === 'mobile' && mobileJoystick.active) {
                    maVoiture.angle = mobileJoystick.angle;
                }

                let utiliseTurbo = keys.turbo && maVoiture.turbo > 0 && keys.avant;
                if (utiliseTurbo) maVoiture.turbo -= 0.6;
                document.getElementById('turbo-bar').style.width = maVoiture.turbo + "%";

                let force = utiliseTurbo ? 1.4 : (keys.arriere ? -0.4 : 0.7);
                if (keys.avant || keys.arriere) { 
                    maVoiture.vx += Math.cos(maVoiture.angle) * force; 
                    maVoiture.vy += Math.sin(maVoiture.angle) * force; 
                }
                maVoiture.vx *= 0.9; maVoiture.vy *= 0.9; maVoiture.x += maVoiture.vx; maVoiture.y += maVoiture.vy;

                // Logique du Bot IA / Synchro Réseau Multi
                if (!estEnLigne) {
                    let diff = Math.min(1.35, 0.45 + (rangIndex * 0.06));
                    if (!isResetting) {
                        bot.angle = Math.atan2(ball.y - bot.y, ball.x - bot.x);
                        bot.vx += Math.cos(bot.angle) * diff; bot.vy += Math.sin(bot.angle) * diff;
                    }
                    bot.vx *= 0.9; bot.vy *= 0.9; bot.x += bot.vx; bot.y += bot.vy;
                } else if(conn && conn.open) {
                    conn.send({ type: 'SYNC_INPUTS', x: maVoiture.x, y: maVoiture.y, angle: maVoiture.angle, turbo: maVoiture.turbo, skin: skinEquipe, boostColor: catalogueBoosts[boostEquipe].couleur });
                }

                if (!estEnLigne || jeSuisHote) {
                    ball.x += ball.vx; ball.y += ball.vy; ball.vx *= 0.98; ball.vy *= 0.98;
                    boostPads.forEach(p => {
                        if (p.active) {
                            if (Math.hypot(joueur.x - p.x, joueur.y - p.y) < 45) { joueur.turbo = Math.min(100, joueur.turbo + 40); p.active = false; p.cooldown = 400; }
                            if (Math.hypot(bot.x - p.x, bot.y - p.y) < 45) { bot.turbo = Math.min(100, bot.turbo + 40); p.active = false; p.cooldown = 400; }
                        }
                    });

                    let hW = terrain.width / 2, hH = terrain.height / 2;
                    if (Math.abs(ball.y) > hH - ball.radius) { ball.vy = -ball.vy * 0.8; ball.y = Math.sign(ball.y) * (hH - ball.radius); }
                    if (Math.abs(ball.x) > hW - ball.radius) {
                        if (Math.abs(ball.y) < 180 && !isResetting) {
                            if (ball.x > 0) { scoreBleu++; declencherBut("BUT POUR TOI !", true); } 
                            else { scoreRouge++; declencherBut("BUT ADVERSE !", false); }
                        } else { ball.vx = -ball.vx * 0.8; ball.x = Math.sign(ball.x) * (hW - ball.radius); }
                    }

                    const col = (c) => {
                        let hx = ball.x - c.x, hy = ball.y - c.y, d = Math.sqrt(hx*hx + hy*hy);
                        if (d < ball.radius + 25) { let a = Math.atan2(hy, hx); ball.vx = Math.cos(a) * 9; ball.vy = Math.sin(a) * 9; }
                    };
                    col(joueur); col(bot);

                    if(estEnLigne && conn && conn.open) conn.send({ type: 'SYNC_BALL', x: ball.x, y: ball.y, sb: scoreBleu, sr: scoreRouge, temps: tempsRestant });
                }

                let vueCam = jeSuisHote || !estEnLigne ? joueur : bot;
                ctx.fillStyle = '#0f172a'; ctx.fillRect(0, 0, canvas.width, canvas.height);
                ctx.save(); ctx.translate(canvas.width / 2 - vueCam.x, canvas.height / 2 - vueCam.y);

                let hW = terrain.width / 2, hH = terrain.height / 2;
                ctx.fillStyle = '#065f46'; ctx.fillRect(-hW, -hH, terrain.width, terrain.height);
                ctx.strokeStyle = 'rgba(255,255,255,0.4)'; ctx.lineWidth = 5; ctx.strokeRect(-hW, -hH, terrain.width, terrain.height);
                ctx.beginPath(); ctx.moveTo(0, -hH); ctx.lineTo(0, hH); ctx.stroke();
                ctx.beginPath(); ctx.arc(0, 0, 150, 0, Math.PI*2); ctx.stroke();
                ctx.lineWidth = 8; ctx.strokeStyle = '#00aaff'; ctx.strokeRect(-hW - 40, -180, 40, 360); ctx.strokeStyle = '#f43f5e'; ctx.strokeRect(hW, -180, 40, 360);

                boostPads.forEach(p => {
                    ctx.beginPath(); ctx.arc(p.x, p.y, 16, 0, Math.PI*2);
                    if (p.active) { ctx.fillStyle = '#facc15'; ctx.fill(); } else { ctx.fillStyle = 'rgba(71,85,105,0.3)'; ctx.fill(); }
                });

                dessinerVoiture(joueur.x, joueur.y, joueur.angle, joueur.skin, null, (jeSuisHote || !estEnLigne ? utiliseTurbo : keys.turbo), joueur.boostColor);
                dessinerVoiture(bot.x, bot.y, bot.angle, bot.skin, estEnLigne ? null : '#f43f5e', (!jeSuisHote && estEnLigne ? utiliseTurbo : false), bot.boostColor);
                dessinerBallon(ball.x, ball.y, ball.radius, angleBalle);
                ctx.restore();
            }
            requestAnimationFrame(update);
        }
        update();
